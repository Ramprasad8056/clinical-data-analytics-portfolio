-- Create products table
CREATE TABLE products (
    product_id INT PRIMARY KEY,
    product_name VARCHAR(100),
    category VARCHAR(50)
);

-- Create orders table
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    order_date DATE,
    region VARCHAR(20),
    product_id INT,
    sales DECIMAL(10,2),
    quantity INT,
    profit DECIMAL(10,2),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);
